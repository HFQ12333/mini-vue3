```javascript
/^[\t\r\n\f ]+/; // 匹配空格、换行等
/^<\/?([a-z][^\t\r\n\f />]*)/i; // 匹配标签名 第一个是<tag,第二个是 tag
/^[^\t\r\n\f />][^\t\r\n\f />=]*/; // 匹配属性名
/^[^\t\r\n\f >]+/; // 匹配属性值
/^[\t\r\n\f ]*=/; // 匹配等号或者前面是空格、换行的等号
/^(v-[A-Za-z0-9-]|:|\.|@|#)/; // 匹配是否是vue中的指令
/(?:^v-([a-z0-9-]+))?(?:(?::|^\.|^@|^#)(\[[^\]]+\]|[^\.]+))?(.+)?$/i; // 匹配指令,指定的参数
```
